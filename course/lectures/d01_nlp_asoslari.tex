% [A-2] MORFOLOGIYA
\begin{frame}{Morfologiya: O'zbek so'z tuzilishi NLP ni qiyinlashtiradi}
  \begin{tcolorbox}[defbox,title=Ta'rif: Morfologiya (so'z tuzilishi)]
    \small
    \textbf{Agglutinativ til}: bitta so'zga bir nechta qo'shimcha yopishadi.
    O'zbek tili --- agglutinativ til;
    ingliz --- izolyativga yaqin (qo'shimchalar kam).
  \end{tcolorbox}
  \vspace{3pt}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{``kitoblarimizdan'' tahlili:}\\[4pt]
      \small
      \begin{tabular}{ll}
        \toprule
        Morfema & Ma'nosi \\
        \midrule
        kitob   & ildiz (ot) \\
        +lar    & ko'plik \\
        +imiz   & I shaxs, ko'plik egalik \\
        +dan    & chiqish kelishigi \\
        \midrule
        \multicolumn{2}{l}{\footnotesize inglizcha: ``from our books'' (3 so'z!)} \\
        \bottomrule
      \end{tabular}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{tcolorbox}[warnbox,title=BoW va TF-IDF muammosi,top=2pt,bottom=2pt]
        \footnotesize
        BoW da har bir shakl \textbf{alohida token} hisoblanadi:\\[3pt]
        \texttt{"kitob"} $\ne$ \texttt{"kitoblar"}\\
        $\ne$ \texttt{"kitoblarimizdan"}\\[3pt]
        $\Rightarrow$ lug'at ($|V|$) portlashi\\
        $\Rightarrow$ siyrak matritsa ($\uparrow$)\\[4pt]
        \textbf{Yechim}: stemming ---\\
        bugungi darsning 2-rejasi.
      \end{tcolorbox}
    \end{column}
  \end{columns}
\end{frame}

% [A-3] SINTAKSIS VA SEMANTIKA
\begin{frame}{Sintaksis va semantika: tartib muhim, ma'no murakkab}

        O'zbek: \textbf{S--O--V}\\

        Ingliz: \textbf{S--V--O}\\

      \end{tcolorbox}
      \vspace{3pt}
      \footnotesize
      POS teglash (L5) so'z turkumini\\
      gap tartibiga asoslab aniqlaydi.
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Semantika} (ma'no):\\[3pt]
      \begin{tcolorbox}[defbox,title=Ko'p ma'nolilik (polysemy),top=2pt,bottom=2pt]
        \footnotesize
        ``\textbf{bank}'' ---\\
        \quad daryo qirg'og'i? \textbf{yoki}\\
        \quad moliya muassasasi?\\[3pt]
        TF-IDF bu farqni \alert{bilmaydi} ---\\
        kontekstsiz son.\\[3pt]
        \textcolor{deepgreen}{\textbf{Yechim}}: Word2Vec, GloVe\\

      \end{tcolorbox}
    \end{column}
  \end{columns}
  \vspace{4pt}
  \begin{tcolorbox}[okbox,title=Kurs manzarasi,top=2pt,bottom=2pt]
    \footnotesize

% [G3] BOW RASMIY TA'RIF
% ----------------------------------------------------------------
\begin{frame}{Bag of Words}
  \begin{tcolorbox}[defbox,title=Ta'rif: Bag of Words]
    \textbf{Lug'at}: $V = \{w_1, w_2, \ldots, w_{|V|}\}$ ---
    korpusdagi barcha unikal tokenlar to'plami.\\[5pt]
    \textbf{gap (dokument) vektori}: $\mathbf{d}_i \in \mathbb{R}^{|V|}$,
    bunda $d_{i,j} = $ ``$w_j$ so'zining $D_i$ gap (dokument)dagi
    uchrash soni''.
  \end{tcolorbox}
  \vspace{5pt}
  \bunda{%
    $V$ = lug'at; $|V|$ = lug'at hajmi;
    $\mathbf{d}_i$ = $i$-gap (dokument) vektori;
    $d_{i,j}$ = $j$-so'z chastotasi $D_i$ da
  }
  \vspace{5pt}
  \textbf{Muhim xossalar:}\\[3pt]
  \small
  \begin{itemize}\setlength\itemsep{3pt}
    \item $d_{i,j} \geq 0$ \quad (chastota manfiy bo'lmaydi)
    \item $d_{i,j} = 0$ \quad ($w_j$ so'zi $D_i$ da yo'q)
    \item Vektor \textbf{siyrak} (sparse): aksariyat qiymatlar $= 0$
    \item So'z tartibi, kontekst, ma'no --- \alert{yo'qoladi}
  \end{itemize}
\end{frame}

% ----------------------------------------------------------------
% [H3a] TF FORMULA
% ----------------------------------------------------------------
\begin{frame}{TF: Gapdagi so'z chastotasi (Term frequency)}
  \begin{tcolorbox}[defbox,title=Ta'rif: TF --- atama chastotasi]
    \[
      \mathrm{TF}(t,\, d)
        = \text{(}t\text{ so'zining } d \text{ gapdagi uchrash soni)}
    \]
  \end{tcolorbox}
  \vspace{4pt}
  \bunda{%
    $t$ = so'z (term); $d$ = gap (dokument);
    bu kursda \textbf{raw count} (normalizatsiyasiz) ishlatiadi
  }
  \vspace{5pt}
  \textbf{Misollar (bizning korpus):}
  \begin{center}\small
  \begin{tabular}{llcc}
    \toprule
    Hujjat & Matn & So'z & $\mathrm{TF}$ \\
    \midrule
    D1 & \texttt{"nlp qiziq"} & nlp & $1$ \\
    D1 & \texttt{"nlp qiziq"} & qiziq & $1$ \\
    D2 & \texttt{"python foydali"} & python & $1$ \\
    D3 & \texttt{"nlp foydali"} & foydali & $1$ \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{3pt}
  \footnotesize\color{halfgray}
  Boshqa ta'riflar ham bor (masalan, normalized: $\mathrm{TF}/|d|$).
  Bu kursda qo'lda hisoblash osonroq bo'lganligi uchun raw count ishlatiladi.
\end{frame}

% ----------------------------------------------------------------
% [H3b] IDF FORMULA
% ----------------------------------------------------------------
\begin{frame}{IDF (Inverse Document Frequency): Noyob so'zlarni mukofotlash}
  \begin{tcolorbox}[defbox,title=Ta'rif: IDF --- teskari gap (dokument) chastotasi]
    \[
      \mathrm{IDF}(t)
        = \ln\!\left(\frac{N}{\mathrm{df}(t)}\right)
    \]
  \end{tcolorbox}
  \vspace{4pt}
  \bunda{%
    $N$ = korpusdagi jami gap (dokument)lar soni;
    $\mathrm{df}(t)$ = $t$ uchraydigan gap (dokument)lar soni;
    $\ln$ = natural logarifm
  }
  \vspace{4pt}
  \textbf{Nima uchun logarifm?}
  \begin{itemize}\small\setlength\itemsep{3pt}
    \item Agar $N/\mathrm{df}$ juda katta bo'lsa (noyob so'z),
          $\ln$ shkalani boshqaruvchan qiladi.
    \item $\mathrm{df}(t) = N$: $\mathrm{IDF} = \ln(1) = 0$
          \quad\textcolor{deepred}{(barcha gap (dokument)larda --- muhim emas)}
    \item $\mathrm{df}(t) = 1$: $\mathrm{IDF} = \ln(N)$
          \quad\textcolor{deepgreen}{(faqat bitta gap (dokument)da --- juda muhim)}
  \end{itemize}
  \vspace{3pt}
  \textbf{TF-IDF}: $\;\mathrm{TF\text{-}IDF}(t, d) = \mathrm{TF}(t,d) \times \mathrm{IDF}(t)$
\end{frame}

\begin{frame}{S1: Logarifm va IDF hisoblash}

    $\ln(a/b) = \ln a - \ln b$
  \end{tcolorbox}
  \vspace{5pt}
  \textbf{Keng qo'llaniladigan qiymatlar:}
  \begin{center}\small
  \begin{tabular}{cccc}
    \toprule
    $N$ & $\mathrm{df}$ & $N/\mathrm{df}$ & $\mathrm{IDF} = \ln(N/\mathrm{df})$ \\
    \midrule
    3 & 1 & 3.00 & $\ln(3) \approx 1.099$ \\
    3 & 2 & 1.50 & $\ln(1.5) \approx 0.405$ \\
    3 & 3 & 1.00 & $\ln(1) = 0.000$ \\
    10 & 1 & 10.0 & $\ln(10) \approx 2.303$ \\
    10 & 5 & 2.00 & $\ln(2) \approx 0.693$ \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{4pt}

  \texttt{import numpy as np; np.log(1.5)}
\end{frame}

\begin{frame}{S2: BoW matritsasini qurish algoritmi}
  \begin{enumerate}\small\setlength\itemsep{5pt}
    \item Barcha gap (dokument)larni tokenizatsiya + preprocessing qiling
    \item Barcha unikal tokenlarni yig'ing $\to$ lug'at $V$,
          ularni tartiblang (alfavit bo'yicha)
    \item $|D| \times |V|$ o'lchamli nol matritsa yarating
    \item Har gap (dokument) uchun: har tokenni $V$ da toping $\to$
          tegishli katakni qiymatini oshiring
  \end{enumerate}
  \vspace{4pt}
  \textbf{Bizning misolimiz:}
  \begin{center}\small
  \begin{tabular}{lcccc}
    \toprule
    Hujjat & foydali & nlp & python & qiziq \\
    \midrule
    D1: ``nlp qiziq''      & 0 & 1 & 0 & 1 \\
    D2: ``python foydali'' & 1 & 0 & 1 & 0 \\
    D3: ``nlp foydali''    & 1 & 1 & 0 & 0 \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{3pt}
  \footnotesize Sklearn: \texttt{CountVectorizer().fit\_transform(corpus).toarray()}
\end{frame}
